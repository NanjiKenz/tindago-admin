function extractFigmaParams(url: string) {
  const match = url.match(/design\/([^\/]+).*node-id=([^&]+)/);
  return {
    fileKey: match?.[1] || '',
    nodeId: match?.[2]?.replace('%3A', ':') || ''
  };
}
